MOD_NAME=年代科技加成
MOD_AUTHOR=玩家
MOD_VERSION=1.0
MOD_DESC=2030年后科技速度翻倍，2050年后再翻倍
#EXEC newyear

# 定义基础倍率变量
VAR TECH_BONUS=1.0

# 2030年后科技翻倍
IF YEAR >= 2030 THEN
    VAR TECH_BONUS=2.0
    PRINT 2030年已到，科技速度翻倍！
    SET BASE_TECH_SPEED=${TECH_BONUS}
END IF

# 2050年后科技再翻倍
IF YEAR >= 2050 THEN
    VAR TECH_BONUS=4.0
    PRINT 2050年已到，科技速度再次翻倍！
    SET BASE_TECH_SPEED=${TECH_BONUS}
END IF
